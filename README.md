## Usage

```js
import AnimatedButton from "animated-button";

function App() {
  return (
    <div>
      <AnimatedButton
        primary
        sml
        loading=false
        text="Click Me"
        handleClick={() => console.log("Clicked me!")}
      ></AnimatedButton>
    </div>
  );
}

export default App;
```
